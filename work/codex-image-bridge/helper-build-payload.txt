#!/usr/bin/env python3
"""Build JSON payload for image bridge from CLI args.

Usage: build-payload.py <out_path> <prompt> <size> <n> [<input_path>...]
"""
import sys
import json
import base64
import mimetypes
from pathlib import Path


def main():
    if len(sys.argv) < 5:
        print("usage: build-payload.py <out> <prompt> <size> <n> [<input>...]", file=sys.stderr)
        sys.exit(2)
    out, prompt, size, n, *inputs = sys.argv[1:]
    body = {"model": "gpt-image-2", "prompt": prompt, "n": int(n), "size": size}
    if inputs:
        body["images"] = []
        for f in inputs:
            p = Path(f)
            if not p.exists():
                print(f"input file missing: {f}", file=sys.stderr)
                sys.exit(3)
            mime, _ = mimetypes.guess_type(str(p))
            if not mime:
                mime = "image/jpeg"
            b64 = base64.b64encode(p.read_bytes()).decode("ascii")
            body["images"].append({"image_url": f"data:{mime};base64,{b64}"})
    Path(out).write_text(json.dumps(body), encoding="utf-8")


if __name__ == "__main__":
    main()
